91 // Helper: build an EntityTemplate from a SpriteAnimDef
92 static EntityTemplate makeFromDef(const Vec2f &initialPos,
93 const SpriteAnimDef &d)
94 {
95 EntityTemplate t;
96 t.id = d.id;
97 t.position = initialPos;
98 t.rotation = 0.0f;
99 t.scale = d.scale;
100
101 t.withVelocity = true;
102 t.velocity.speed = d.speed;
103 t.velocity.direction = d.direction;
104
105 t.sprite.texturePath = d.texturePath;
106 t.sprite.rectLeft = d.rectLeft;
107 t.sprite.rectTop = d.rectTop;
108 t.sprite.rectWidth = d.rectWidth;
109 t.sprite.rectHeight = d.rectHeight;
110 t.sprite.zIndex = d.zIndex;
111 t.sprite.red = d.red;
112 t.sprite.green = d.green;
113 t.sprite.blue = d.blue;
114
115 t.withAnimation = d.withAnimation;
116 if (d.withAnimation) {
117 t.animation.frameWidth = d.frameWidth;
118 t.animation.frameHeight = d.frameHeight;
119 t.animation.frameLeft = d.frameLeft;
120 t.animation.frameTop = d.frameTop;
121 t.animation.totalFrames = d.totalFrames;
122 }
123 return t;
124 }

126 // Helper: build an EntityTemplate for a parallax layer
127 static EntityTemplate makeParallax(const ParallaxDef &p)
128 {
129 EntityTemplate t;
130 t.id = "parallax_layer";
131 t.position = {0.0f, 0.0f};
132 t.rotation = 0.0f;
133 t.scale = p.scale;
134
135 t.withVelocity = true;
136 t.velocity.speed = p.velocitySpeed;
137 t.velocity.direction = p.velocityDir;
138
139 t.sprite.texturePath = p.texturePath;
140 t.sprite.rectLeft = 0;
141 t.sprite.rectTop = 0;
142 t.sprite.rectWidth = p.rectWidth;
143 t.sprite.rectHeight = p.rectHeight;
144 t.sprite.zIndex = -1000;
145 t.sprite.red = 255;
146 t.sprite.opacity = p.opacity;
147
148 t.withParallax = true;
149 t.parallax.textureWidth = p.textureWidth;
150 t.parallax.speed = p.parallaxSpeed;
151 return t;
152 }
